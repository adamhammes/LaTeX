\begin{align*}
	P(I | D) &= \dfrac{ P(D | I)P(I) }{P(D | I)P(I) + P(D | \overline{I} )P(\overline{I})}\\
	&= \dfrac{0.05 * .2}{ 0.05 * 0.2 + 0.3 * 0.8}\\

\end{align*}
