\section*{Problem 2}
\begin{enumerate}[(a)]
	\item
	\begin{align*}
		P(A|B) &= \dfrac{P(A \cap B)}{P(B)} \\
		&= \dfrac{.5}{.75} \\
		&= \dfrac{2}{3}
	\end{align*}
	
	\item
	$P(A) = 0.65$, and $P(A|B) = 2/3 \neq 0.65$; therefore the occurrence of $B$ has an effect on $P(A)$ and the events are dependent.
	
	\item
	\begin{align*}
		P(B|A) &= \dfrac{P(B \cap A)}{P(A)} \\
		&= \dfrac{0.5}{0.65} \\
		&= \dfrac{10}{13}
	\end{align*}


\end{enumerate}
